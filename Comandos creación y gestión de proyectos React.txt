// Instalar React Router 
npm i react-router

// Renderizar <BrowserRouter> en la aplicación

import {
  createBrowserRouter,
  RouterProvider,
} from "react-router";


let router = createBrowserRouter([
  {
    path: "/",
    Component: Root,
    children: [
      {
        path: "shows/:showId",
        element: <Show />
      },
    ],
  },
]);

createRoot(document.getElementById("root")).render(
  <StrictMode>
    <RouterProvider router={router} />
  </StrictMode>
);
